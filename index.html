How the Code Works

  1. Simulates a JOIN in JavaScript using two data structures.
  2. Finds module and its prerequisite by matching codes to names.
  3. Appends each match as a new row in the HTML table.
<script>
const MODULES = {
  M101: "Programming Basics",
  M102: "Data Structures",
  M103: "Algorithms",
  M201: "Web Development I",
  M202: "Web Development II",
  M301: "AI Applications",
  M302: "Linear Algebra"
};

const PRE_REQUISITES = [
  { mod_code: "M102", requisite: "M101" },
  { mod_code: "M103", requisite: "M102" },
  { mod_code: "M202", requisite: "M201" },
  { mod_code: "M301", requisite: "M302" }
];

const tableBody = document.querySelector("#resultTable tbody");

PRE_REQUISITES.forEach(record => {
  const modName = MODULES[record.mod_code];
  const prereqName = MODULES[record.requisite];

  if (modName && prereqName) {
    const row = `
      ${modName}
      ${prereqName}
    `;
    tableBody.innerHTML += row;
  }
});
</script>
    
SELECT m1.mod_name AS Module, m2.mod_name AS Prerequisite
FROM module m1
JOIN pre_requisite p ON m1.mod_code = p.mod_code
JOIN module m2 ON p.requisite = m2.mod_code;
